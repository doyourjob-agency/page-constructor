## Parameters

`type: 'highlight-table-block'`

`title?: string` — Title on the left

`description?: string` — Description on the left

`contentSize?: ContentSize (s | l)` - Content size for table & legend.

`legend?: string[]` — Legend for the table

`table: HighlightTable` — HighlightTable content

-   `content: (string | HighlightTableCell)[][]` — Text in the table cells

-   -   `cell: string` — Text in the table cell
-   -   `buttonUrl?: string` — Url button in cell
-   -   `buttonText?: string` — Text button in cell
-   -   `columnWidth?: 12 | 6 | 4 | 3 | 2` — Custom column width for cell

-   `highlighter?: string[]` — Colors in the table rows

-   `customColumnWidth?: (12 | 6 | 4 | 3 | 2)[]` — Sizes in the table cells

-   `justify?: ('left' | 'center' | 'right')[]` — Text alignment.

</StoryTemplate>
